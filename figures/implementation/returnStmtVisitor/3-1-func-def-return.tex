public void visit(Func_defNode node) {
    symtable.openScope();
    node.getBlockNode().accept(this);
    if (node.getType() != Type.VOID && (!blockNodeHasReturnStmt())) {
        errorHandler.returnNotGuaranteed();
    }
    symtable.closeScope();
}
