\begin{lstlisting}[caption={A small recursive program, printing the first 20 numbers of the fibonacci sequence}, label={ex02}]
func int fib(int fibNum){
  if(fibNum < 3){
    return 1
  }
  else{
    return fib(fibNum - 1) + fib(fibNum - 2)
  }
}

func Setup(){
  int i
  i := 1
  while(i <= 20){
    Print(String(fib(i)))
    i := i + 1
  }
}
\end{lstlisting}
