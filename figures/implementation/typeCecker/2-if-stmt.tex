$$@Override
public void visit(If_stmtNode node) {
    node.getExpr().accept(this);
    node.getIfBlock().accept(this);
    if (node.getElseBlock() != null) {
        node.getElseBlock().accept(this);
    }
    checkSpecificType(node.getExpr(), Type.BOOL);
}
